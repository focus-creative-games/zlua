nextlua = nextlua or {}

nextlua.typeof = function(t)
    return __nextlua_typeof(t)
end

nextlua.make_generic_type = function(generic_type, ...)
    return __nextlua_make_generic_type(generic_type, ...)
end

nextlua.create_signature = function(method_name, ...)
    return __nextlua_create_signature(method_name, ...)
end

nextlua.signature = nextlua.create_signature

nextlua.get_method = function(obj, signature)
    return obj[signature]
end

nextlua.register_method = function(obj, name, method)
    local mt = getmetatable(obj)
    if mt == nil then
        error("nextlua.register_method requires userdata with metatable")
    end
    mt[name] = method
end

local function resolve_corlib_type(type_name)
    local assemblies = {
        CSharp.mscorlib,
        CSharp.System_Private_CoreLib,
        CSharp.netstandard,
    }

    for i = 1, #assemblies do
        local asm = assemblies[i]
        if asm ~= nil then
            local t = asm[type_name]
            if t ~= nil then
                return t
            end
        end
    end
    return nil
end

local function safe_typeof(type_table)
    if type_table == nil then
        return nil
    end
    local ok, ret = pcall(nextlua.typeof, type_table)
    if ok then
        return ret
    end
    return nil
end

nextlua.corlibtypes = {
    int32 = safe_typeof(resolve_corlib_type("System.Int32")),
    int64 = safe_typeof(resolve_corlib_type("System.Int64")),
    string = safe_typeof(resolve_corlib_type("System.String")),
    boolean = safe_typeof(resolve_corlib_type("System.Boolean")),
    object = safe_typeof(resolve_corlib_type("System.Object")),
}

return nextlua
