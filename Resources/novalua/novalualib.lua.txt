novalua = novalua or {}

novalua.typeof = function(t)
    return __novalua_typeof(t)
end

novalua.make_generic_type = function(generic_type, ...)
    return __novalua_make_generic_type(generic_type, ...)
end

novalua.create_signature = function(method_name, ...)
    return __novalua_create_signature(method_name, ...)
end

novalua.signature = novalua.create_signature

novalua.get_method = function(obj, signature)
    return obj[signature]
end

novalua.register_method = function(obj, name, method)
    local mt = getmetatable(obj)
    if mt == nil then
        error("novalua.register_method requires userdata with metatable")
    end
    mt[name] = method
end

local function resolve_corlib_type(type_name)
    local assemblies = {
        CSharp.mscorlib,
        CSharp.System_Private_CoreLib,
        CSharp.netstandard,
    }

    for i = 1, #assemblies do
        local asm = assemblies[i]
        if asm ~= nil then
            local t = asm[type_name]
            if t ~= nil then
                return t
            end
        end
    end
    return nil
end

local function safe_typeof(type_table)
    if type_table == nil then
        return nil
    end
    local ok, ret = pcall(novalua.typeof, type_table)
    if ok then
        return ret
    end
    return nil
end

novalua.corlibtypes = {
    int32 = safe_typeof(resolve_corlib_type("System.Int32")),
    int64 = safe_typeof(resolve_corlib_type("System.Int64")),
    string = safe_typeof(resolve_corlib_type("System.String")),
    boolean = safe_typeof(resolve_corlib_type("System.Boolean")),
    object = safe_typeof(resolve_corlib_type("System.Object")),
}

return novalua
